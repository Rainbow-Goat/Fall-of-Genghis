		id = 100000
		title = "EVTNAME100000"
		desc = "EVTDESC100000"
		
		trigger = {
			tag = ENG

			NOT = {
				has_global_flag = norman_win_1000
			}

			}

			set_country_flag = englaland_rules_waves
		}
}
